395 su2double dotProd(const CSysVector & u, const CSysVector & v) {
396 
397  /*--- check for consistent sizes ---*/
398  if (u.nElm != v.nElm) {
399  cerr << "CSysVector friend dotProd(CSysVector, CSysVector): "
400  << "CSysVector sizes do not match";
401  throw(-1);
402  }
403 
404  /*--- find local inner product and, if a parallel run, sum over all
405  processors (we use nElemDomain instead of nElem) ---*/
406  su2double loc_prod = 0.0;
407  for (unsigned long i = 0; i < u.nElmDomain; i++)
408  loc_prod += u.vec_val[i]*v.vec_val[i];
409  su2double prod = 0.0;
410 
411 #ifdef HAVE_MPI
412  SU2_MPI::Allreduce(&loc_prod, &prod, 1, MPI_DOUBLE, MPI_SUM, MPI_COMM_WORLD);
413 #else
414  prod = loc_prod;
415 #endif
416 
417  return prod;
