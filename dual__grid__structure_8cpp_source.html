423  unsigned short iDim;
424  su2double vec_a[3] = {0.0,0.0,0.0}, vec_b[3] = {0.0,0.0,0.0}, vec_c[3] = {0.0,0.0,0.0}, vec_d[3] = {0.0,0.0,0.0}, Local_Volume;
425 
426  AD::StartPreacc();
427  AD::SetPreaccIn(val_coord_Edge_CG, nDim);
428  AD::SetPreaccIn(val_coord_Elem_CG, nDim);
429  AD::SetPreaccIn(val_coord_FaceElem_CG, nDim);
430  AD::SetPreaccIn(val_coord_Point, nDim);
431 
432  for (iDim = 0; iDim < nDim; iDim++) {
433  vec_a[iDim] = val_coord_Edge_CG[iDim]-val_coord_Point[iDim];
434  vec_b[iDim] = val_coord_FaceElem_CG[iDim]-val_coord_Point[iDim];
435  vec_c[iDim] = val_coord_Elem_CG[iDim]-val_coord_Point[iDim];
436  }
437 
438  vec_d[0] = vec_a[1]*vec_b[2]-vec_a[2]*vec_b[1];
439  vec_d[1] = -(vec_a[0]*vec_b[2]-vec_a[2]*vec_b[0]);
440  vec_d[2] = vec_a[0]*vec_b[1]-vec_a[1]*vec_b[0];
441 
442  Local_Volume = fabs(vec_c[0]*vec_d[0] + vec_c[1]*vec_d[1] + vec_c[2]*vec_d[2])/6.0;
443 
444  AD::SetPreaccOut(Local_Volume);
445  AD::EndPreacc();
446 
447  return Local_Volume;
448 }
