580 void CWaveSolver::ImplicitEuler_Iteration(CGeometry *geometry, CSolver **solver_container, CConfig *config) {
581 
582  unsigned short iVar;
583  unsigned long iPoint, total_index;
584 
585  /*--- Set maximum residual to zero ---*/
586 
587  for (iVar = 0; iVar < nVar; iVar++) {
588  SetRes_RMS(iVar, 0.0);
589  SetRes_Max(iVar, 0.0, 0);
590  }
591 
592  /*--- Build implicit system ---*/
593 
594  for (iPoint = 0; iPoint < geometry->GetnPointDomain(); iPoint++) {
595 
596  /*--- Right hand side of the system (-Residual) and initial guess (x = 0) ---*/
597 
598  for (iVar = 0; iVar < nVar; iVar++) {
599  total_index = iPoint*nVar+iVar;
600  LinSysSol[total_index] = 0.0;
601  AddRes_RMS(iVar, LinSysRes[total_index]*LinSysRes[total_index]);
602  AddRes_Max(iVar, fabs(LinSysRes[total_index]), geometry->node[iPoint]->GetGlobalIndex(), geometry->node[iPoint]->GetCoord());
603  }
604  }
605 
606  /*--- Initialize residual and solution at the ghost points ---*/
607 
608  for (iPoint = geometry->GetnPointDomain(); iPoint < geometry->GetnPoint(); iPoint++) {
609  for (iVar = 0; iVar < nVar; iVar++) {
610  total_index = iPoint*nVar + iVar;
611  LinSysRes[total_index] = 0.0;
612  LinSysSol[total_index] = 0.0;
613  }
614  }
615 
616  /*--- Solve or smooth the linear system ---*/
617 
618  CSysSolve system;
619  system.Solve(Jacobian, LinSysRes, LinSysSol, geometry, config);
620 
621  /*--- Update solution (system written in terms of increments) ---*/
622 
623  for (iPoint = 0; iPoint < geometry->GetnPointDomain(); iPoint++) {
624  for (iVar = 0; iVar < nVar; iVar++) {
625  node[iPoint]->AddSolution(iVar, LinSysSol[iPoint*nVar+iVar]);
626  }
627  }
628 
629  /*--- MPI solution ---*/
630 
631  Set_MPI_Solution(geometry, config);
632 
633  /*--- Compute the root mean square residual ---*/
634 
635  SetResidual_RMS(geometry, config);
636
