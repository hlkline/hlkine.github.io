917 
918  AD::StartPreacc();
919  AD::SetPreaccIn(Coord_i, nDim); AD::SetPreaccIn(Coord_j, nDim);
920  AD::SetPreaccIn(Normal, nDim);
921  AD::SetPreaccIn(TurbVar_Grad_i, nVar, nDim); AD::SetPreaccIn(TurbVar_Grad_j, nVar, nDim);
922  AD::SetPreaccIn(F1_i); AD::SetPreaccIn(F1_j);
923 
924  if (incompressible) {
925  AD::SetPreaccIn(V_i, nDim+5); AD::SetPreaccIn(V_j, nDim+5);
926 
927  Density_i = V_i[nDim+1]; Density_j = V_j[nDim+1];
928  Laminar_Viscosity_i = V_i[nDim+3]; Laminar_Viscosity_j = V_j[nDim+3];
929  Eddy_Viscosity_i = V_i[nDim+4]; Eddy_Viscosity_j = V_j[nDim+4];
930  }
931  else {
932  AD::SetPreaccIn(V_i, nDim+7); AD::SetPreaccIn(V_j, nDim+7);
933 
934  Density_i = V_i[nDim+2]; Density_j = V_j[nDim+2];
935  Laminar_Viscosity_i = V_i[nDim+5]; Laminar_Viscosity_j = V_j[nDim+5];
936  Eddy_Viscosity_i = V_i[nDim+6]; Eddy_Viscosity_j = V_j[nDim+6];
937  }
938 
939  /*--- Compute the blended constant for the viscous terms ---*/
940  sigma_kine_i = F1_i*sigma_k1 + (1.0 - F1_i)*sigma_k2;
941  sigma_kine_j = F1_j*sigma_k1 + (1.0 - F1_j)*sigma_k2;
942  sigma_omega_i = F1_i*sigma_om1 + (1.0 - F1_i)*sigma_om2;
943  sigma_omega_j = F1_j*sigma_om1 + (1.0 - F1_j)*sigma_om2;
944 
945  /*--- Compute mean effective viscosity ---*/
946  diff_i_kine = Laminar_Viscosity_i + sigma_kine_i*Eddy_Viscosity_i;
947  diff_j_kine = Laminar_Viscosity_j + sigma_kine_j*Eddy_Viscosity_j;
948  diff_i_omega = Laminar_Viscosity_i + sigma_omega_i*Eddy_Viscosity_i;
949  diff_j_omega = Laminar_Viscosity_j + sigma_omega_j*Eddy_Viscosity_j;
950 
951  diff_kine = 0.5*(diff_i_kine + diff_j_kine); // Could instead use weighted average!
952  diff_omega = 0.5*(diff_i_omega + diff_j_omega);
953 
954  /*--- Compute vector going from iPoint to jPoint ---*/
955  dist_ij_2 = 0; proj_vector_ij = 0;
956  for (iDim = 0; iDim < nDim; iDim++) {
957  Edge_Vector[iDim] = Coord_j[iDim]-Coord_i[iDim];
958  dist_ij_2 += Edge_Vector[iDim]*Edge_Vector[iDim];
959  proj_vector_ij += Edge_Vector[iDim]*Normal[iDim];
960  }
961  if (dist_ij_2 == 0.0) proj_vector_ij = 0.0;
962  else proj_vector_ij = proj_vector_ij/dist_ij_2;
963 
964  /*--- Mean gradient approximation. Projection of the mean gradient in the direction of the edge ---*/
965  for (iVar = 0; iVar < nVar; iVar++) {
966  Proj_Mean_GradTurbVar_Normal[iVar] = 0.0;
967  Proj_Mean_GradTurbVar_Edge[iVar] = 0.0;
968  for (iDim = 0; iDim < nDim; iDim++) {
969  Mean_GradTurbVar[iVar][iDim] = 0.5*(TurbVar_Grad_i[iVar][iDim] + TurbVar_Grad_j[iVar][iDim]);
970  Proj_Mean_GradTurbVar_Normal[iVar] += Mean_GradTurbVar[iVar][iDim]*Normal[iDim];
971  }
972  Proj_Mean_GradTurbVar_Corrected[iVar] = Proj_Mean_GradTurbVar_Normal[iVar];
973  }
974 
975  val_residual[0] = diff_kine*Proj_Mean_GradTurbVar_Corrected[0];
976  val_residual[1] = diff_omega*Proj_Mean_GradTurbVar_Corrected[1];

1058  sigma_kine_j = F1_j*sigma_k1 + (1.0 - F1_j)*sigma_k2;

1093  }

1096  val_residual[1] = diff_omega*Proj_Mean_GradTurbVar_Corrected[1];

1099  if (implicit) {

1101  Jacobian_i[1][0] = 0.0; Jacobian_i[1][1] = -diff_omega*proj_vector_ij/Density_i;
1102 
1103  Jacobian_j[0][0] = diff_kine*proj_vector_ij/Density_j; Jacobian_j[0][1] = 0.0;
1104  Jacobian_j[1][0] = 0.0; Jacobian_j[1][1] = diff_omega*proj_vector_ij/Density_j;
1105  }
