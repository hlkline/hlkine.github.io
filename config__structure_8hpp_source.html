737  void addDoubleOption(const string name, su2double & option_field, su2double default_value) {
738  // Check if the key is already in the map. If this fails, it is coder error
739  // and not user error, so throw.
740  assert(option_map.find(name) == option_map.end());
741 
742  // Add this option to the list of all the options
743  all_options.insert(pair<string, bool>(name, true));
744 
745  // Create the parser for a su2double option with a reference to the option_field and the desired
746  // default value. This will take the string in the config file, convert it to a su2double, and
747  // place that su2double in the memory location specified by the reference.
748  COptionBase* val = new COptionDouble(name, option_field, default_value);
749 
750  // Create an association between the option name ("CFL") and the parser generated above.
751  // During configuration, the parsing script will get the option name, and use this map
752  // to find how to parse that option.
753  option_map.insert(pair<string, COptionBase *>(name, val));
754  }
