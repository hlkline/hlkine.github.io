320 void CMeanFlowIteration::SetWind_GustField(CConfig *config_container, CGeometry **geometry_container, CSolver ***solver_container) {
321  // The gust is imposed on the flow field via the grid velocities. This method called the Field Velocity Method is described in the
322  // NASA TM–2012-217771 - Development, Verification and Use of Gust Modeling in the NASA Computational Fluid Dynamics Code FUN3D
323  // the desired gust is prescribed as the negative of the grid velocity.
324 
325  // If a source term is included to account for the gust field, the method is described by Jones et al. as the Split Velocity Method in
326  // Simulation of Airfoil Gust Responses Using Prescribed Velocities.
327  // In this routine the gust derivatives needed for the source term are calculated when applicable.
328  // If the gust derivatives are zero the source term is also zero.
329  // The source term itself is implemented in the class CSourceWindGust
330 
331  int rank = MASTER_NODE;
332 #ifdef HAVE_MPI
333  MPI_Comm_rank(MPI_COMM_WORLD, &rank);
334 #endif
335 
336  if (rank == MASTER_NODE)
337  cout << endl << "Running simulation with a Wind Gust." << endl;
338  unsigned short iDim, nDim = geometry_container[MESH_0]->GetnDim(); //We assume nDim = 2
339  if (nDim != 2) {
340  if (rank == MASTER_NODE) {
341  cout << endl << "WARNING - Wind Gust capability is only verified for 2 dimensional simulations." << endl;
342  }
343  }
344 
345  /*--- Gust Parameters from config ---*/
346  unsigned short Gust_Type = config_container->GetGust_Type();
347  su2double xbegin = config_container->GetGust_Begin_Loc(); // Location at which the gust begins.
348  su2double L = config_container->GetGust_WaveLength(); // Gust size
349  su2double tbegin = config_container->GetGust_Begin_Time(); // Physical time at which the gust begins.
350  su2double gust_amp = config_container->GetGust_Ampl(); // Gust amplitude
351  su2double n = config_container->GetGust_Periods(); // Number of gust periods
352  unsigned short GustDir = config_container->GetGust_Dir(); // Gust direction
353 
354  /*--- Variables needed to compute the gust ---*/
355  unsigned short Kind_Grid_Movement = config_container->GetKind_GridMovement(ZONE_0);
356  unsigned long iPoint;
357  unsigned short iMGlevel, nMGlevel = config_container->GetnMGLevels();
358 
359  su2double x, y, x_gust, dgust_dx, dgust_dy, dgust_dt;
360  su2double *Gust, *GridVel, *NewGridVel, *GustDer;
361 
362  su2double Physical_dt = config_container->GetDelta_UnstTime();
363  unsigned long ExtIter = config_container->GetExtIter();
364  su2double Physical_t = ExtIter*Physical_dt;
365 
366  su2double Uinf = solver_container[MESH_0][FLOW_SOL]->GetVelocity_Inf(0); // Assumption gust moves at infinity velocity
367 
368  Gust = new su2double [nDim];
369  NewGridVel = new su2double [nDim];
370  for (iDim = 0; iDim < nDim; iDim++) {
371  Gust[iDim] = 0.0;
372  NewGridVel[iDim] = 0.0;
373  }
374 
375  GustDer = new su2double [3];
376  for (unsigned short i = 0; i < 3; i++) {
377  GustDer[i] = 0.0;
378  }
379 
380  // Vortex variables
381  unsigned long nVortex = 0;
382  vector<su2double> x0, y0, vort_strenth, r_core; //vortex is positive in clockwise direction.
383  if (Gust_Type == VORTEX) {
384  InitializeVortexDistribution(nVortex, x0, y0, vort_strenth, r_core);
385  }
386 
387  /*--- Check to make sure gust lenght is not zero or negative (vortex gust doesn't use this). ---*/
388  if (L <= 0.0 && Gust_Type != VORTEX) {
389  if (rank == MASTER_NODE) cout << "ERROR: The gust length needs to be positive" << endl;
390 #ifndef HAVE_MPI
391  exit(EXIT_FAILURE);
392 #else
393  MPI_Abort(MPI_COMM_WORLD,1);
394  MPI_Finalize();
395 #endif
396  }
397 
398  /*--- Loop over all multigrid levels ---*/
399 
400  for (iMGlevel = 0; iMGlevel <= nMGlevel; iMGlevel++) {
401 
402  /*--- Loop over each node in the volume mesh ---*/
403 
404  for (iPoint = 0; iPoint < geometry_container[iMGlevel]->GetnPoint(); iPoint++) {
405 
406  /*--- Reset the Grid Velocity to zero if there is no grid movement ---*/
407  if (Kind_Grid_Movement == GUST) {
408  for (iDim = 0; iDim < nDim; iDim++)
409  geometry_container[iMGlevel]->node[iPoint]->SetGridVel(iDim, 0.0);
410  }
411 
412  /*--- initialize the gust and derivatives to zero everywhere ---*/
413 
414  for (iDim = 0; iDim < nDim; iDim++) {Gust[iDim]=0.0;}
415  dgust_dx = 0.0; dgust_dy = 0.0; dgust_dt = 0.0;
416 
417  /*--- Begin applying the gust ---*/
418 
419  if (Physical_t >= tbegin) {
420 
421  x = geometry_container[iMGlevel]->node[iPoint]->GetCoord()[0]; // x-location of the node.
422  y = geometry_container[iMGlevel]->node[iPoint]->GetCoord()[1]; // y-location of the node.
423 
424  // Gust coordinate
425  x_gust = (x - xbegin - Uinf*(Physical_t-tbegin))/L;
426 
427  /*--- Calculate the specified gust ---*/
428  switch (Gust_Type) {
429 
430  case TOP_HAT:
431  // Check if we are in the region where the gust is active
432  if (x_gust > 0 && x_gust < n) {
433  Gust[GustDir] = gust_amp;
434  // Still need to put the gust derivatives. Think about this.
435  }
436  break;
437 
438  case SINE:
439  // Check if we are in the region where the gust is active
440  if (x_gust > 0 && x_gust < n) {
441  Gust[GustDir] = gust_amp*(sin(2*PI_NUMBER*x_gust));
442 
443  // Gust derivatives
444  //dgust_dx = gust_amp*2*PI_NUMBER*(cos(2*PI_NUMBER*x_gust))/L;
445  //dgust_dy = 0;
446  //dgust_dt = gust_amp*2*PI_NUMBER*(cos(2*PI_NUMBER*x_gust))*(-Uinf)/L;
447  }
448  break;
449 
450  case ONE_M_COSINE:
451  // Check if we are in the region where the gust is active
452  if (x_gust > 0 && x_gust < n) {
453  Gust[GustDir] = gust_amp*(1-cos(2*PI_NUMBER*x_gust));
454 
455  // Gust derivatives
456  //dgust_dx = gust_amp*2*PI_NUMBER*(sin(2*PI_NUMBER*x_gust))/L;
457  //dgust_dy = 0;
458  //dgust_dt = gust_amp*2*PI_NUMBER*(sin(2*PI_NUMBER*x_gust))*(-Uinf)/L;
459  }
460  break;
461 
462  case EOG:
463  // Check if we are in the region where the gust is active
464  if (x_gust > 0 && x_gust < n) {
465  Gust[GustDir] = -0.37*gust_amp*sin(3*PI_NUMBER*x_gust)*(1-cos(2*PI_NUMBER*x_gust));
466  }
467  break;
468 
469  case VORTEX:
470 
471  /*--- Use vortex distribution ---*/
472  // Algebraic vortex equation.
473  for (unsigned long i=0; i<nVortex; i++) {
474  su2double r2 = pow(x-(x0[i]+Uinf*(Physical_t-tbegin)), 2) + pow(y-y0[i], 2);
475  su2double r = sqrt(r2);
476  su2double v_theta = vort_strenth[i]/(2*PI_NUMBER) * r/(r2+pow(r_core[i],2));
477  Gust[0] = Gust[0] + v_theta*(y-y0[i])/r;
478  Gust[1] = Gust[1] - v_theta*(x-(x0[i]+Uinf*(Physical_t-tbegin)))/r;
479  }
480  break;
481 
482  case NONE: default:
483 
484  /*--- There is no wind gust specified. ---*/
485  if (rank == MASTER_NODE) {
486  cout << "No wind gust specified." << endl;
487  }
488  break;
489 
490  }
491  }
492 
493  /*--- Set the Wind Gust, Wind Gust Derivatives and the Grid Velocities ---*/
494 
495  GustDer[0] = dgust_dx;
496  GustDer[1] = dgust_dy;
497  GustDer[2] = dgust_dt;
498 
499  solver_container[iMGlevel][FLOW_SOL]->node[iPoint]->SetWindGust(Gust);
500  solver_container[iMGlevel][FLOW_SOL]->node[iPoint]->SetWindGustDer(GustDer);
501 
502  GridVel = geometry_container[iMGlevel]->node[iPoint]->GetGridVel();
503 
504  /*--- Store new grid velocity ---*/
505 
506  for (iDim = 0; iDim < nDim; iDim++) {
507  NewGridVel[iDim] = GridVel[iDim] - Gust[iDim];
508  geometry_container[iMGlevel]->node[iPoint]->SetGridVel(iDim, NewGridVel[iDim]);
509  }
510 
511  }
512  }
513 
514  delete [] Gust;
515  delete [] GustDer;
516  delete [] NewGridVel;
