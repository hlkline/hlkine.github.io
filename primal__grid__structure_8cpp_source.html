54 void CPrimalGrid::SetCoord_CG(su2double **val_coord) {

56 
57  AD::StartPreacc();
58  AD::SetPreaccIn(val_coord, GetnNodes(), nDim);
59 
60  for (iDim = 0; iDim < nDim; iDim++) {
61  Coord_CG[iDim] = 0.0;
62  for (iNode = 0; iNode < GetnNodes(); iNode++)
63  Coord_CG[iDim] += val_coord[iNode][iDim]/su2double(GetnNodes());
64  }
65 
66  for (iFace = 0; iFace < GetnFaces(); iFace++)
67  for (iDim = 0; iDim < nDim; iDim++) {
68  Coord_FaceElems_CG[iFace][iDim] = 0.0;
69  for (iNode = 0; iNode < GetnNodesFace(iFace); iNode++) {

72  }
73  }
74 
75  AD::SetPreaccOut(Coord_CG, nDim);
76  AD::SetPreaccOut(Coord_FaceElems_CG, GetnFaces(), nDim);
