731  // take in extra options. The addXxxOptions mostly follow the same pattern, so please see addDoubleOption
732  // for detailed comments.
733  //
734  // List options are those that can be an unknown number of elements, and also take in a reference to
735  // an integer. This integer will be populated with the number of elements of that type unmarshaled.
736  //
737  // Array options are those with a fixed number of elements.
738  //
739  // List and Array options should also be able to be specified with the string "NONE" indicating that there
740  // are no elements. This allows the option to be present in a config file but left blank.

745  void addDoubleOption(const string name, su2double & option_field, su2double default_value) {
746  // Check if the key is already in the map. If this fails, it is coder error
747  // and not user error, so throw.
748  assert(option_map.find(name) == option_map.end());
749 
750  // Add this option to the list of all the options
751  all_options.insert(pair<string, bool>(name, true));
752 
753  // Create the parser for a su2double option with a reference to the option_field and the desired
754  // default value. This will take the string in the config file, convert it to a su2double, and
755  // place that su2double in the memory location specified by the reference.
756  COptionBase* val = new COptionDouble(name, option_field, default_value);
757 
758  // Create an association between the option name ("CFL") and the parser generated above.
759  // During configuration, the parsing script will get the option name, and use this map
760  // to find how to parse that option.
761  option_map.insert(pair<string, COptionBase *>(name, val));
762  }
